```kotlin
package org.fcast.android.sender.capture

import android.hardware.display.DisplayManager
import android.hardware.display.VirtualDisplay
import android.media.projection.MediaProjection
import android.opengl.EGL14
import android.opengl.EGLConfig
import android.opengl.EGLContext
import android.opengl.EGLDisplay
import android.opengl.EGLSurface
import android.opengl.GLES20
import android.os.Handler
import android.os.HandlerThread
import android.util.Log
import androidx.annotation.WorkerThread

/**
 * Owns the EGL/GL pipeline that converts the projected display into YUV planes
 * and ships them to native code.
 *
 * Instantiated by the coordinator (one per capture session); shutdown by the
 * coordinator when the user stops, the projection callback fires, or the
 * activity is destroyed.
 *
 * **Threading:** all GL work happens on the [glThread] HandlerThread. Public
 * `start` / `shutdown` are safe from the main thread.
 */
class CaptureEngine {

    @Volatile private var running = false

    private val glThread = HandlerThread("CaptureEngineGL").also { it.start() }
    private val glHandler = Handler(glThread.looper)

    private var virtualDisplay: VirtualDisplay? = null

    private var eglDisplay: EGLDisplay = EGL14.EGL_NO_DISPLAY
    private var eglContext: EGLContext = EGL14.EGL_NO_CONTEXT
    private var eglSurface: EGLSurface = EGL14.EGL_NO_SURFACE

    private val yFbo = IntArray(1)
    private val uFbo = IntArray(1)
    private val vFbo = IntArray(1)
    private val yTex = IntArray(1)
    private val uTex = IntArray(1)
    private val vTex = IntArray(1)

    private var lastFrameNanos: Long = 0L
    private var minIntervalNanos: Long = 0L

    fun start(
        projection: MediaProjection,
        config: CaptureConfig,
        onStarted: (width: Int, height: Int) -> Unit,
        onFatalError: (reason: String) -> Unit,
    ) {
        check(!running) { "CaptureEngine.start called twice" }
        running = true
        minIntervalNanos = config.minIntervalNanos

        glHandler.post {
            try {
                initEgl()
                val (w, h) = setupVirtualDisplay(projection, config)
                onStarted(w, h)
            } catch (t: Throwable) {
                Log.e(TAG, "GL init failed", t)
                running = false
                onFatalError(t.message ?: "GL init failed")
            }
        }
    }

    fun shutdown() {
        if (!running) return
        running = false
        glHandler.post {
            try {
                virtualDisplay?.release()
                virtualDisplay = null
                releaseGl()
            } catch (t: Throwable) {
                Log.w(TAG, "shutdown raced with a frame", t)
            }
        }
        glThread.quitSafely()
        try { glThread.join(1000L) } catch (_: InterruptedException) {}
    }

    @WorkerThread
    private fun initEgl() {
        eglDisplay = EGL14.eglGetDisplay(EGL14.EGL_DEFAULT_DISPLAY)
        require(eglDisplay != EGL14.EGL_NO_DISPLAY) { "eglGetDisplay failed" }

        val version = IntArray(2)
        EGL14.eglInitialize(eglDisplay, version, 0, version, 1)

        val configs = arrayOfNulls<EGLConfig>(1)
        val numConfig = IntArray(1)
        EGL14.eglChooseConfig(
            eglDisplay,
            EGL_CONFIG_ATTRIBS, 0, configs, 0, 1, numConfig, 0
        )

        val ctxAttribs = intArrayOf(EGL14.EGL_CONTEXT_CLIENT_VERSION, 2, EGL14.EGL_NONE)
        eglContext = EGL14.eglCreateContext(
            eglDisplay, configs[0], EGL14.EGL_NO_CONTEXT, ctxAttribs, 0
        )

        val pbufAttribs = intArrayOf(EGL14.EGL_WIDTH, 1, EGL14.EGL_HEIGHT, 1, EGL14.EGL_NONE)
        eglSurface = EGL14.eglCreatePbufferSurface(eglDisplay, configs[0], pbufAttribs, 0)
        require(EGL14.eglMakeCurrent(eglDisplay, eglSurface, eglSurface, eglContext)) {
            "eglMakeCurrent failed: " + EGL14.eglGetError()
        }

        // …allocate yTex/uTex/vTex and yFbo/uFbo/vFbo verbatim from the
        // existing MainActivity code. Out of scope for the skeleton.
        Log.i(TAG, "EGL initialised")
    }

    @WorkerThread
    private fun setupVirtualDisplay(
        projection: MediaProjection,
        config: CaptureConfig,
    ): Pair<Int, Int> {

        // and binds the VirtualDisplay to it. Copy that code verbatim here.

    }

    /** Per-frame entry; called by the SurfaceTexture's onFrameAvailable listener. */
    @WorkerThread
    fun onFrameAvailable() {
        if (!running) return
        val now = System.nanoTime()
        if (now - lastFrameNanos < minIntervalNanos) return
        lastFrameNanos = now
        pumpOneFrame()
    }

    @WorkerThread
    fun pumpOneFrame() {

        //   - render Y, U, V into the three FBOs
        //   - glReadPixels into three ByteBuffers

    @WorkerThread
    private fun releaseGl() {
        if (yFbo[0] != 0) { GLES20.glDeleteFramebuffers(3, intArrayOf(yFbo[0], uFbo[0], vFbo[0]), 0); yFbo[0] = 0; uFbo[0] = 0; vFbo[0] = 0 }
        if (yTex[0] != 0) { GLES20.glDeleteTextures(3, intArrayOf(yTex[0], uTex[0], vTex[0]), 0);     yTex[0] = 0; uTex[0] = 0; vTex[0] = 0 }
        if (eglSurface != EGL14.EGL_NO_SURFACE) { EGL14.eglDestroySurface(eglDisplay, eglSurface); eglSurface = EGL14.EGL_NO_SURFACE }
        if (eglContext != EGL14.EGL_NO_CONTEXT) { EGL14.eglDestroyContext(eglDisplay, eglContext); eglContext = EGL14.EGL_NO_CONTEXT }
        if (eglDisplay != EGL14.EGL_NO_DISPLAY) { EGL14.eglTerminate(eglDisplay); eglDisplay = EGL14.EGL_NO_DISPLAY }
    }

    companion object {
        private const val TAG = "CaptureEngine"

        private val EGL_CONFIG_ATTRIBS = intArrayOf(
            EGL14.EGL_RENDERABLE_TYPE, EGL14.EGL_OPENGL_ES2_BIT,
            EGL14.EGL_RED_SIZE,        8,
            EGL14.EGL_GREEN_SIZE,      8,
            EGL14.EGL_BLUE_SIZE,       8,
            EGL14.EGL_ALPHA_SIZE,      8,
            EGL14.EGL_DEPTH_SIZE,      0,
            EGL14.EGL_STENCIL_SIZE,    0,
            EGL14.EGL_NONE
        )
    }
}
```

## What's `TODO` and why

`setupVirtualDisplay` and `pumpOneFrame` are deliberately left as `TODO` in
this skeleton. Their *implementations* live on `main` inside `MainActivity`
and must be **copied verbatim** during step 04's sub-PR 3 (engine move).

Reviewers should diff the moved bodies against the originals byte-for-byte —
this step is a code-move, not a rewrite. Any behavioural change goes in step
12 (performance pass).

## Threading invariants

- Every method tagged `@WorkerThread` runs on `glThread`.
- `start` / `shutdown` are safe from any thread but are designed to be called
  from main.
- The `running` flag is `@Volatile` and is the single source of truth for
  "is the engine alive". The coordinator's `engine != null` mirrors this.

## Native-method call site

`pumpOneFrame()` ends with a call to native code:

```kotlin
MainActivity.nativeProcessFrame(/* …existing args… */)
```

The `nativeProcessFrame` declaration stays on `MainActivity` (this step does
not move JNI symbols). Step 07 / 08 move it as part of the broader JNI
re-organisation.

If `MainActivity.nativeProcessFrame` is package-private (default), exposing
it from a `capture/` subpackage works because Kotlin can call package-private
Java members in the same package. If not, change its visibility to `public`
in the `MainActivity.java` patch.
