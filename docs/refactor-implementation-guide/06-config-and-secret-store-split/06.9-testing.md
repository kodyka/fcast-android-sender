## Sample migration test

Add to `src/config/migration.rs`:

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::app;
    use crate::secret::InMemorySecretStore;

    fn fresh_app() {
        let app = app::App::production()
            .with_secrets(Box::new(InMemorySecretStore::new()));
        // `OnceCell::set` returns Err the second time ‚Äî but that's fine for tests
        // that share a process. For a per-test isolated App, refactor App::init to
        // accept a `&'static Mutex<Option<App>>` injection point.
        let _ = app::init(app);
    }

    #[test]
    fn extracts_plaintext() {
        fresh_app();
        let tmp = tempfile::NamedTempFile::new().unwrap();
        std::fs::write(
            tmp.path(),
            r#"{"gstpop_api_key":"secret-123","endpoint":"https://example.com/v1"}"#,
        ).unwrap();
        migrate_config_file(tmp.path()).unwrap();
        let raw = std::fs::read_to_string(tmp.path()).unwrap();
        assert!(!raw.contains("secret-123"));
        assert!(raw.contains("gstpop.api_key.v1"));
        let bytes = crate::app::app().secrets().get(DEFAULT_ALIAS).unwrap();
        assert_eq!(bytes.as_str().unwrap(), "secret-123");
    }

    #[test]
    fn idempotent() {
        fresh_app();
        let tmp = tempfile::NamedTempFile::new().unwrap();
        std::fs::write(
            tmp.path(),
            r#"{"gstpop_api_key_alias":"gstpop.api_key.v1","endpoint":"https://example.com/v1"}"#,
        ).unwrap();
        migrate_config_file(tmp.path()).unwrap();
        let raw = std::fs::read_to_string(tmp.path()).unwrap();
        assert!(raw.contains("gstpop.api_key.v1"));
    }
}
```

Depends on `tempfile = { version = "3", features = [] }` in `[dev-dependencies]`.

## Manual on-device smoke

```bash
adb shell run-as org.fcast.android.sender cat /data/data/org.fcast.android.sender/files/migration-config.json
```

After step 06 lands, the file should not contain a `gstpop_api_key` field ‚Äî
only `gstpop_api_key_alias`. If you see the plaintext field, the migration
did not run; check `adb logcat -d | rg 'migration config'`.

## Encryption verification

```bash
adb shell run-as org.fcast.android.sender ls -la shared_prefs/
adb shell run-as org.fcast.android.sender cat shared_prefs/fcast_secrets.xml | head
```

The `fcast_secrets.xml` content should be opaque encrypted base64 ‚Äî not
human-readable. If you can read your API key in plaintext, the
`EncryptedSharedPreferences` setup is wrong.
